`, styles: [`.print-area { width: 800px; padding: 16px; }`] }) export class GeneratePdfComponent { @ViewChild('printSection', { static: false }) printSection!: ElementRef; async downloadPdf() { const element = this.printSection.nativeElement as HTMLElement; // Wait for fonts to load (important for correct Chinese rendering) try { await (document as any).fonts?.ready; } catch (e) { /* ignore */ } const canvas = await html2canvas(element, { scale: 2, // higher scale => higher-res image useCORS: true, allowTaint: false, logging: false }); const imgData = canvas.toDataURL('image/png'); const pdf = new jsPDF('p', 'mm', 'a4'); const pdfWidth = pdf.internal.pageSize.getWidth(); const pdfHeight = pdf.internal.pageSize.getHeight(); const imgProps = (pdf as any).getImageProperties(imgData); const imgWidthMm = pdfWidth; const imgHeightMm = (imgProps.height * imgWidthMm) / imgProps.width; let heightLeft = imgHeightMm; let position = 0; // Add first page pdf.addImage(imgData, 'PNG', 0, position, imgWidthMm, imgHeightMm); heightLeft -= pdfHeight; // Add remaining pages while (heightLeft > 0) { position = heightLeft - imgHeightMm; pdf.addPage(); pdf.addImage(imgData, 'PNG', 0, position, imgWidthMm, imgHeightMm); heightLeft -= pdfHeight; } pdf.save('document-image.pdf'); } }
