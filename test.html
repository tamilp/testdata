function findMaxMin(arr) { // Edge case: if the array is empty, return null for both max and min if (arr.length === 0) { return { max: null, min: null }; } // Initialize max and min to the first element of the array let max = arr[0]; let min = arr[0]; // Iterate through the array starting from the second element for (let i = 1; i < arr.length; i++) { // Update max if current element is greater than max if (arr[i] > max) { max = arr[i]; } // Update min if current element is smaller than min if (arr[i] < min) { min = arr[i]; } } // Return an object with max and min return { max, min }; } // Example usage const arr = [4, 7, 1, 9, 2, 6]; const result = findMaxMin(arr); console.log(`Max: ${result.max}, Min: ${result.min}`);
