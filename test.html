/** * Sort an array of objects by label in given locale * @param {Array} arr - Array of objects { label, key } * @param {string} locale - Language locale code, e.g., 'en', 'ta', 'zh' * @param {string} order - 'asc' for ascending, 'desc' for descending * @returns {Array} - Sorted array */ function sortByLabel(arr, locale = 'en', order = 'asc') { return arr.sort((a, b) => { const result = a.label.localeCompare(b.label, locale, { sensitivity: 'base' }); return order === 'asc' ? result : -result; }); }
