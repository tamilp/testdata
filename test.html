// Replace 'my-component' with your actual custom element tag const host = document.querySelector('my-component'); // If shadowRoot is open const shadowRoot = host.shadowRoot; // Select your fixed button inside the shadow DOM const button = shadowRoot.querySelector('.fixed-button'); if (button) { const rect = button.getBoundingClientRect(); const viewportCenterY = window.innerHeight / 2; const buttonCenterY = rect.top + rect.height / 2; const diff = Math.round(buttonCenterY - viewportCenterY); console.log("Button vertical center Y:", Math.round(buttonCenterY)); console.log("Viewport center Y:", Math.round(viewportCenterY)); console.log("➡️ Difference from vertical center:", diff + "px"); if (Math.abs(diff) <= 2) { console.log("✅ Button is centered (within ±2px tolerance)"); } else { console.log("❌ Button is off-center by", diff, "px"); } } else { console.warn("Button not found inside the shadow DOM."); }
