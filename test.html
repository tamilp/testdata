function toCamelCase(str) { return str .toLowerCase() // Convert everything to lowercase .split('_') // Split by underscore .map((word, index) => index === 0 ? word : word.charAt(0).toUpperCase() + word.slice(1) ) // Capitalize first letter of each word except the first one .join(''); // Join words back } console.log(toCamelCase("TEST_TESTING")); // Output: "testTesting"
