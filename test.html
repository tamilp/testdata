import { HTTP_INTERCEPTORS } from '@angular/common/http'; import { LoggingInterceptor } from './logging.interceptor'; @NgModule({ providers: [ { provide: HTTP_INTERCEPTORS, useClass: LoggingInterceptor, multi: true } ] }) export class AppModule {} import { Injectable } from '@angular/core'; import { HttpInterceptor, HttpRequest, HttpHandler, HttpEvent, HttpEventType } from '@angular/common/http'; import { Observable, tap } from 'rxjs'; @Injectable() export class DetailedLoggingInterceptor implements HttpInterceptor { intercept(req: HttpRequest, next: HttpHandler): Observable> { console.log(`[REQUEST START] ${req.method} ${req.urlWithParams}`); const started = Date.now(); return next.handle(req).pipe( tap({ next: (event: HttpEvent) => { switch (event.type) { case HttpEventType.Sent: console.log(`[REQUEST SENT to backend] ${req.method} ${req.urlWithParams}`); break; case HttpEventType.ResponseHeader: console.log(`[RESPONSE STARTED] headers received for ${req.method} ${req.urlWithParams}`); break; case HttpEventType.DownloadProgress: console.log(`[RESPONSE IN PROGRESS] ${req.method} ${req.urlWithParams}, received: ${event.loaded} bytes`); break; case HttpEventType.UploadProgress: console.log(`[REQUEST UPLOAD IN PROGRESS] ${req.method} ${req.urlWithParams}, sent: ${event.loaded} bytes`); break; case HttpEventType.Response: const elapsed = Date.now() - started; console.log(`[RESPONSE COMPLETE] ${req.method} ${req.urlWithParams} (${elapsed} ms)`, event.body); break; } }, error: (error) => { const elapsed = Date.now() - started; console.error(`[RESPONSE ERROR] ${req.method} ${req.urlWithParams} (${elapsed} ms)`, error); } }) ); } }
