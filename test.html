function documentUpload(url: string, data?: any, httpOptions?: any): Promise { const headers = (httpOptions && httpOptions.headers) || {}; const started = Date.now(); const timestamp = () => new Date().toISOString(); // Request start console.log(`[${timestamp()}] [FETCH REQUEST START] URL: ${url}`); return fetch(url, { method: 'POST', headers, body: data }) .then(response => { // Request end (fetch has sent request, got a response object) console.log(`[${timestamp()}] [FETCH REQUEST END] URL: ${url}`); // Response start (we have status + headers) console.log(`[${timestamp()}] [FETCH RESPONSE START] URL: ${url} Status: ${response.status}`); return response.json().then(body => { const elapsed = Date.now() - started; // Response end (body fully read) console.log(`[${timestamp()}] [FETCH RESPONSE END] URL: ${url} Time: ${elapsed} ms`); return body; }); }) .catch(error => { const elapsed = Date.now() - started; console.error(`[${timestamp()}] [FETCH REQUEST END WITH ERROR] URL: ${url} Time: ${elapsed} ms`, error); throw error; }); }
