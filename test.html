function documentUpload(url: string, data?: any, httpOptions?: any): Promise { const headers = (httpOptions && httpOptions.headers) || {}; const started = Date.now(); const timestamp = () => new Date().toISOString(); const elapsed = () => `${Date.now() - started} ms`; // Request start console.log(`[${timestamp()} | +${elapsed()}] [FETCH REQUEST START] URL: ${url}`); return fetch(url, { method: 'POST', headers, body: data }) .then(response => { // Request end console.log(`[${timestamp()} | +${elapsed()}] [FETCH REQUEST END] URL: ${url}`); // Response start (status + headers available) console.log(`[${timestamp()} | +${elapsed()}] [FETCH RESPONSE START] URL: ${url} Status: ${response.status}`); return response.json().then(body => { // Response end (body fully read) console.log(`[${timestamp()} | +${elapsed()}] [FETCH RESPONSE END] URL: ${url}`); return body; }); }) .catch(error => { console.error(`[${timestamp()} | +${elapsed()}] [FETCH REQUEST END WITH ERROR] URL: ${url}`, error); throw error; }); }
