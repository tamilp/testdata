scrollTabs(direction: 'left' | 'right'): void { const tabListEl = this.tabList.nativeElement; const scrollAmount = 150; if (direction === 'left') { tabListEl.scrollBy({ left: -scrollAmount, behavior: 'smooth' }); } else { tabListEl.scrollBy({ left: scrollAmount, behavior: 'smooth' }); } } private scrollToActiveTab(): void { const tabListEl = this.tabList.nativeElement; const activeTabEl = tabListEl.querySelector('.active'); if (activeTabEl) { tabListEl.scrollTo({ left: activeTabEl.offsetLeft - tabListEl.clientWidth / 2 + activeTabEl.clientWidth / 2, behavior: 'smooth' }); } }
