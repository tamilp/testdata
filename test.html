import { HttpEvent, HttpInterceptorFn, HttpRequest, HttpHandlerFn, HttpEventType } from '@angular/common/http'; import { tap } from 'rxjs'; export const detailedLoggingInterceptor: HttpInterceptorFn = (req: HttpRequest, next: HttpHandlerFn) => { const started = performance.now(); const formatTime = () => `${(performance.now() - started).toFixed(3)} ms`; console.log(`[${formatTime()}] [REQUEST START] ${req.method} ${req.urlWithParams}`); return next(req).pipe( tap({ next: (event: HttpEvent) => { switch (event.type) { case HttpEventType.Sent: console.log(`[${formatTime()}] [REQUEST SENT to backend] ${req.method} ${req.urlWithParams}`); break; case HttpEventType.ResponseHeader: console.log(`[${formatTime()}] [RESPONSE STARTED] headers received for ${req.method} ${req.urlWithParams}`); break; case HttpEventType.DownloadProgress: console.log(`[${formatTime()}] [RESPONSE IN PROGRESS] ${req.method} ${req.urlWithParams}, received: ${event.loaded} bytes`); break; case HttpEventType.UploadProgress: console.log(`[${formatTime()}] [REQUEST UPLOAD IN PROGRESS] ${req.method} ${req.urlWithParams}, sent: ${event.loaded} bytes`); break; case HttpEventType.Response: console.log(`[${formatTime()}] [RESPONSE COMPLETE] ${req.method} ${req.urlWithParams}`, event.body); break; } }, error: (error) => { console.error(`[${formatTime()}] [RESPONSE ERROR] ${req.method} ${req.urlWithParams}`, error); } }) ); };
